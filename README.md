**📋 Setup & Deployment**
Prerequisites
bash
pip install -r requirements.txt
aws configure  # AWS credentials
Local Development
bash
# Clone repo
git clone https://github.com/Mayur-cyber/Team-Invincible-Ryzen-project-Pasmi-AWS-.git

cd Team-Invincible-Ryzen-project-Pasmi-AWS-

# Install dependencies
pip install -r requirements.txt

# Run locally
streamlit run app.py
AWS Deployment
bash
# Deploy serverless stack
cdk deploy --all

# Or use provided CloudFormation
aws cloudformation deploy --template-file infrastructure.yaml

🔧 Configuration

# config.yaml
bedrock:
  model: "anthropic.claude-3-sonnet-20240229-v1:0"
  max_tokens: 1000

social_apis:
 
  youtube: "YOUR_YOUTUBE_API_KEY"
  
  Instagram: "YOUR_INSTAGRAM_ACCESS_TOKEN"
  
  Facebook: "YOUR_FB_PAGE_TOKEN"
  
  Twitter: "YOUR_TWITTER_BEARER_TOKEN"
